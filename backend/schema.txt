CREATE TABLE public.analytics_clicks (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  scheduled_post_id uuid,
  source_platform text,
  tracked_url text,
  CONSTRAINT analytics_clicks_pkey PRIMARY KEY (id),
  CONSTRAINT analytics_clicks_scheduled_post_id_fkey FOREIGN KEY (scheduled_post_id) REFERENCES public.scheduled_posts(id)
);
CREATE TABLE public.analytics_events (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  created_at timestamp without time zone DEFAULT now(),
  link_id uuid,
  user_agent text,
  ip_address text,
  referrer text,
  platform text,
  CONSTRAINT analytics_events_pkey PRIMARY KEY (id),
  CONSTRAINT analytics_events_link_id_fkey FOREIGN KEY (link_id) REFERENCES public.tracked_links(id)
);
CREATE TABLE public.businesses (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  name text NOT NULL,
  description text,
  website_url text,
  user_id uuid NOT NULL,
  CONSTRAINT businesses_pkey PRIMARY KEY (id),
  CONSTRAINT businesses_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
CREATE TABLE public.email_templates (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  name text NOT NULL,
  subject text NOT NULL,
  body text NOT NULL,
  CONSTRAINT email_templates_pkey PRIMARY KEY (id)
);
CREATE TABLE public.generated_posts (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  user_prompt text,
  caption text,
  hashtags ARRAY,
  image_url text NOT NULL,
  image_prompt text,
  user_id uuid NOT NULL,
  CONSTRAINT generated_posts_pkey PRIMARY KEY (id),
  CONSTRAINT generated_posts_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
CREATE TABLE public.leads (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  email text NOT NULL UNIQUE,
  name text,
  source text,
  score integer NOT NULL DEFAULT 0,
  last_contacted_at timestamp with time zone,
  CONSTRAINT leads_pkey PRIMARY KEY (id)
);
CREATE TABLE public.meta_accounts (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  business_id uuid,
  system_token text NOT NULL,
  facebook_page_id text,
  facebook_page_username text,
  page_access_token text,
  page_token_expires_at timestamp with time zone,
  instagram_business_account_id text,
  instagram_username text,
  meta jsonb,
  created_at timestamp with time zone DEFAULT now(),
  updated_at timestamp with time zone DEFAULT now(),
  CONSTRAINT meta_accounts_pkey PRIMARY KEY (id),
  CONSTRAINT meta_accounts_business_id_fkey FOREIGN KEY (business_id) REFERENCES public.businesses(id)
);
CREATE TABLE public.scheduled_posts (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  generated_post_id uuid NOT NULL,
  social_account_id uuid NOT NULL,
  scheduled_time timestamp with time zone NOT NULL,
  status text NOT NULL DEFAULT 'pending'::text,
  posted_at timestamp with time zone,
  platform_post_id text,
  error_message text,
  CONSTRAINT scheduled_posts_pkey PRIMARY KEY (id),
  CONSTRAINT scheduled_posts_generated_post_id_fkey FOREIGN KEY (generated_post_id) REFERENCES public.generated_posts(id),
  CONSTRAINT scheduled_posts_social_account_id_fkey FOREIGN KEY (social_account_id) REFERENCES public.social_accounts(id)
);
CREATE TABLE public.social_accounts (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  platform text NOT NULL,
  account_name text NOT NULL,
  account_id text,
  access_token text NOT NULL,
  token_expires_at timestamp with time zone,
  business_id uuid,
  refresh_token text,
  token_type text,
  scopes ARRAY,
  meta jsonb DEFAULT '{}'::jsonb,
  is_active boolean DEFAULT true,
  updated_at timestamp with time zone DEFAULT now(),
  CONSTRAINT social_accounts_pkey PRIMARY KEY (id),
  CONSTRAINT social_accounts_business_id_fkey FOREIGN KEY (business_id) REFERENCES public.businesses(id)
);
CREATE TABLE public.tracked_links (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  created_at timestamp without time zone DEFAULT now(),
  original_url text NOT NULL,
  short_code text NOT NULL UNIQUE,
  scheduled_post_id uuid,
  user_id uuid,
  platform character varying,
  CONSTRAINT tracked_links_pkey PRIMARY KEY (id),
  CONSTRAINT tracked_links_scheduled_post_id_fkey FOREIGN KEY (scheduled_post_id) REFERENCES public.scheduled_posts(id),
  CONSTRAINT tracked_links_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
